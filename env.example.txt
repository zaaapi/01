# Variáveis de Ambiente - LIVIA

# Supabase Configuration
# Obtenha essas informações no dashboard do Supabase: https://app.supabase.com
# Settings > API

# Project URL (exemplo: https://abcdefghijklmnop.supabase.co)
NEXT_PUBLIC_SUPABASE_URL=https://seu-projeto.supabase.co

# Anon Public Key (use a chave "anon" "public", NÃO a "service_role")
NEXT_PUBLIC_SUPABASE_ANON_KEY=sua-chave-anon-key-aqui

# Instruções Detalhadas:
# 1. Acesse https://app.supabase.com
# 2. Selecione seu projeto
# 3. Vá em Settings > API
# 4. Copie a "Project URL" e cole acima
# 5. Copie a chave "anon" "public" (NÃO use service_role)
# 6. Cole acima substituindo "sua-chave-anon-key-aqui"
# 7. Salve este arquivo como .env.local na raiz do projeto
# 8. Reinicie o servidor de desenvolvimento (npm run dev)
